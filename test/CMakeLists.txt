set(THIS Fraction-Lib-Tests)

enable_testing()

    FractionTests.cpp
)

target_link_libraries(${THIS}
                        gtest_main
                        Fraction-Lib
)

add_test(
    NAME ${THIS}
    COMMAND ${THIS}
)
